    内置函数部分解析¶

    • help()

    查看内置模块的使用帮助

    • slice()、reversed()、zip()
    _str = "abc"
_tuple = (1,2,3)
_list = [1,2,3]


# slice()
_list[slice(0,2)]  # 等价于：_list[0:2]


# reversed()
# 反转后得到一个迭代器
r = reversed(x)  # <reversed object at 0x105acab50>
[i for i in r]  # [3,2,1]


# zip()
r = zip(_tuple,_list)  # <zip object at 0x10777b550>
[i for i in r]  # [(1, 1), (2, 2), (3, 3)]
    • enumerate()
    lst = ['one','two','three','four','five']
[i for i in enumerate(lst)]  # [(0, 'one'), (1, 'two'), (2, 'three'), (3, 'four'), (4, 'five')]
[i for i in enumerate(lst, 1)]  # 标号从1开始
    • all()、any()
    all(Iterable)  # Iterable元素中全是True，则返回True
any(Iterable)  # Iterable元素中有一个是True，则返回True
    • input()、print()
    r = input("请输入...")  # 返回str类型


# 注意：
# 1. 同时输出多个值，默认空格分隔，可以自定义seq变量控制
# 2. 输出结束会末尾会自动加一个换行，可以自定义end变量控制
print(1，2，3， seq="", end="\n")
    • eval()、exec()、compile()

    执行字符串形式的表达式

    e1 = "1+2"
e2 = "[1,2,3]"

eval(e1)  # 3
eval(e2)  # [1,2,3]

    执行字符串形式的语句

    s = """
li = []
for i in range(5):
    li.append(i)
"""

exec(s)  # None，exec执行但不会返回值

    执行交互式语句

    p = "print(123)"
i = "input('请输入')"

eval(p)  # 123
exec(p)  # 123

r = eval(i)  # 请输入
print(r)  # 返回输入值

r = exec(i)  # 请输入
print(exec(i))  # 返回None

    compile(string, "", mode)可以将字符串编译为eval()和exec()可执行的格式

    code1 = "1+2"
com1 = compile(code1, "", mode="eval")
print(eval(com1))  # 3

code2 = "for i in range(3): print(i)"
com2 = compile(code2, "", mode="exec")
exec(com2)   # 1 2 3


code3 = "name = input('请输入:')"  # 输入:hello
# 编译为交互形式
com3 = compile(code3, "", mode="single")
# 执行编译对象，eval/exec执行都可以
eval(com3)
exec(com3)
name = ""
print(name)  # hello
